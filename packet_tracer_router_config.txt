# Cisco Packet Tracer Wireless Router Configuration
# CLI commands for router setup

# Step 1: Basic wireless configuration
enable
configure terminal

# Step 2: Configure wireless interface
interface Dot11Radio0
 ssid MeshNetwork
 authentication open
 authentication key-management wpa version 2
 wpa-psk ascii 0 mesh123

# Step 3: Configure IP and DHCP
interface Vlan1
 ip address 192.168.0.1 255.255.255.0
 no shutdown

ip dhcp pool MESH_POOL
 network 192.168.0.0 255.255.255.0
 default-router 192.168.0.1
 dns-server 8.8.8.8

# Step 4: Enable wireless
interface Dot11Radio0
 no shutdown

exit
exit
write memory
